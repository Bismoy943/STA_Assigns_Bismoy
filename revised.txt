WITH scb AS (
    SELECT 
        branch_code,
        country_code,
        ods,
        MAX(sci_entity_code) AS sci_entity_code
    FROM fdp_mfu_staging_dev.m_scb_entity_xlate
    WHERE ods='2025-06-30'
      AND tp_system='OTP'
    GROUP BY branch_code, country_code, ods
),

oaf_glel_trd_dedup AS (
    SELECT *
    FROM (
        SELECT *,
               ROW_NUMBER() OVER (PARTITION BY link_id, country, ods ORDER BY link_id) rn
        FROM fdp_otp_all_sri_open_dev.oaf_glel_trd
        WHERE ods='2025-06-30' AND country='LK'
    ) t WHERE rn = 1
),

oaf_glel_bal_dedup AS (
    SELECT *
    FROM (
        SELECT *,
               ROW_NUMBER() OVER (PARTITION BY link_id, country, ods ORDER BY link_id) rn
        FROM fdp_otp_all_sri_open_dev.oaf_glel_bal
        WHERE ods='2025-06-30' AND country='LK'
    ) t WHERE rn = 1
),

parl_dedup AS (
    SELECT *
    FROM (
        SELECT 
            SPLIT(regexp_replace(trim(ACCT_ID), ' ', ''), '[|]')[0] AS acct_clean,
            PRTY_ID,
            CTRY_CD,
            ODS_VAL AS ods,
            ROW_NUMBER() OVER (
                PARTITION BY 
                    SPLIT(regexp_replace(trim(ACCT_ID), ' ', ''), '[|]')[0],
                    CTRY_CD,
                    ODS_VAL
                ORDER BY PRTY_ID DESC
            ) rn
        FROM fdp_daas_dev.PRTY_ACCT_RLTD
        WHERE ods_val='2025-06-30'
          AND CTRY_CD='LK'
    ) t WHERE rn = 1
)
select count(*) as ct FROM 
(SELECT ACCTID.entity_code as entity_code,ACCTID.ods as ods,substring(oaf_glel_bal.transaction_branch_code, 3, 3) AS key1,CASE WHEN length(ACCTID.ACCT_ID)=12 THEN substring(ACCTID.ACCT_ID,1,3) WHEN length(ACCTID.ACCT_ID)=10 THEN substring(ACCTID.ACCT_ID,3,3) ELSE NULL END AS key2,substring(basel_txn.branch_code, 3, 3) AS key3,substring(otp_scbt_t_ips_basel_txn.branch_code, 3, 3) AS key4 
FROM (SELECT DISTINCT date(ods) AS ods, country AS entity_code, link_id AS ACCT_ID, NULL AS fdp_is_rollover_flag, NULL AS fdp_last_processed_date 
FROM fdp_otp_all_sri_open_dev.oaf_glel_trd A WHERE A.ods='2025-06-30' and A.country='LK'
UNION SELECT DISTINCT date(ods) AS ods, country AS entity_code, link_id AS ACCT_ID, NULL AS fdp_is_rollover_flag, NULL AS fdp_last_processed_date FROM fdp_otp_all_sri_open_dev.oaf_glel_bal B WHERE B.ods='2025-06-30' and B.country='LK' 
UNION SELECT DISTINCT date(ods) AS ods, country AS entity_code, deal_reference AS ACCT_ID, NULL AS fdp_is_rollover_flag, NULL AS fdp_last_processed_date FROM fdp_otp_all_sri_open_dev.oaf_glel_trd C WHERE C.ods='2025-06-30' and C.country='LK' 
UNION SELECT DISTINCT date(PARL.ODS_VAL) AS ods, PARL.CTRY_CD AS entity_code, SPLIT(regexp_replace(trim(PARL.ACCT_ID), ' ', ''), '[|]')[0] AS ACCT_ID, NULL AS fdp_is_rollover_flag, NULL AS fdp_last_processed_date FROM fdp_daas_dev.PRTY_ACCT_RLTD PARL WHERE PARL.ods_val='2025-06-30' and PARL.CTRY_CD='LK' AND process_id_val='PRTY_ACCT_RLTD-OTP-TPCUST_JRNL' 
UNION SELECT DISTINCT date(ods) AS ods, case when country_code='SP' then 'SP' else country end AS entity_code, CASE WHEN otp_dsa_slpd_bs.fieldname27 IN ('FINA', 'IAIR', 'IIRA', 'ODFA', 'NINC', 'CIRA', 'CFIN') 
AND otp_dsa_slpd_bs.bic_c_refid=otp_dsa_slpd_bs.fieldname35 THEN concat(otp_dsa_slpd_bs.bic_c_refidco, '+F') WHEN otp_dsa_slpd_bs.fieldname27='BANL' AND otp_dsa_slpd_bs.bic_c_refid=otp_dsa_slpd_bs.fieldname35 THEN concat(otp_dsa_slpd_bs.bic_c_refidco, '+C') 
WHEN otp_dsa_slpd_bs.fieldname27='IPIA' AND otp_dsa_slpd_bs.bic_c_refid=otp_dsa_slpd_bs.fieldname35 THEN concat(otp_dsa_slpd_bs.bic_c_refidco, '+I') WHEN otp_dsa_slpd_bs.fieldname27='RPIP' AND otp_dsa_slpd_bs.bic_c_refid=otp_dsa_slpd_bs.fieldname35 THEN concat(otp_dsa_slpd_bs.bic_c_refidco, '+R') 
WHEN otp_dsa_slpd_bs.fieldname27 IN ('FREC', 'UNAT') AND fieldname24 IS NULL THEN otp_dsa_slpd_bs.bic_c_refidco ELSE otp_dsa_slpd_bs.fieldname35 END AS ACCT_ID, NULL AS fdp_is_rollover_flag, NULL AS fdp_last_processed_date FROM fdp_dfds_otp_all_sri_open_dev.otp_fpsl_accounting_bs otp_dsa_slpd_bs WHERE ods='2025-06-30' and country='LK' 
UNION SELECT DISTINCT date(ods) AS ods, case when country_code='SP' then 'SP' else country end AS entity_code, CASE WHEN TRIM(cast(otp_dsa_slpd_pl.bic_c_refidco AS varchar(128))) IS NOT NULL AND fieldname27 IN ('IINC', 'CREV') THEN Concat(otp_dsa_slpd_pl.bic_c_refidco, '+F') 
WHEN TRIM(cast(otp_dsa_slpd_pl.bic_c_refidco AS varchar(128))) IS NOT NULL AND fieldname27 NOT IN ('IINC', 'CREV') THEN (cast(otp_dsa_slpd_pl.bic_c_refidco AS varchar(128))) ELSE bic_c_refid END AS ACCT_ID, NULL AS fdp_is_rollover_flag, NULL AS fdp_last_processed_date FROM fdp_dfds_otp_all_sri_open_dev.otp_fpsl_accounting_pl otp_dsa_slpd_pl WHERE ods='2025-06-30' and country='LK' 
UNION SELECT DISTINCT date(ods) AS ods, country AS entity_code, txn_id AS ACCT_ID, NULL AS fdp_is_rollover_flag, NULL AS fdp_last_processed_date FROM fdp_otp_all_sri_open_dev.otp_scbt_t_ips_basel_txn WHERE ods='2025-06-30' and country='LK' 
UNION SELECT DISTINCT date(ods) AS ods, case when country_code='SP' then 'SP' else country end AS entity_code, bic_c_refid AS ACCT_ID, NULL AS fdp_is_rollover_flag, NULL AS fdp_last_processed_date FROM fdp_dfds_otp_all_sri_open_dev.otp_fpsl_accounting_bs otp_bs WHERE ods='2025-06-30' and country='LK' 
UNION SELECT DISTINCT date(ods) AS ods, case when country_code='SP' then 'SP' else country end AS entity_code, bic_c_refid AS ACCT_ID, NULL AS fdp_is_rollover_flag, NULL AS fdp_last_processed_date FROM fdp_dfds_otp_all_sri_open_dev.otp_fpsl_accounting_pl WHERE ods='2025-06-30' and country='LK' ) ACCTID
LEFT OUTER JOIN oaf_glel_trd_dedup oaf_glel_trd
ON ACCTID.ACCT_ID = oaf_glel_trd.link_id
AND ACCTID.entity_code = oaf_glel_trd.country
AND ACCTID.ods = oaf_glel_trd.ods
LEFT OUTER JOIN oaf_glel_bal_dedup oaf_glel_bal
ON oaf_glel_trd.link_id = oaf_glel_bal.link_id
AND oaf_glel_trd.country = oaf_glel_bal.country
AND oaf_glel_trd.ods = oaf_glel_bal.ods
LEFT OUTER JOIN parl_dedup PARLTP
ON PARLTP.acct_clean = ACCTID.ACCT_ID
AND PARLTP.CTRY_CD = ACCTID.entity_code
AND PARLTP.ods = ACCTID.ods
LEFT OUTER JOIN scb
ON (
    substring(oaf_glel_bal.transaction_branch_code, 3, 3) = scb.branch_code
    OR
    CASE 
        WHEN length(ACCTID.ACCT_ID)=12 THEN substring(ACCTID.ACCT_ID,1,3)
        WHEN length(ACCTID.ACCT_ID)=10 THEN substring(ACCTID.ACCT_ID,3,3)
    END = scb.branch_code
)
AND ACCTID.entity_code = scb.country_code
AND ACCTID.ods = scb.ods
WHERE ACCTID.ods='2025-06-30'
AND ACCTID.entity_code='LK'
) base
